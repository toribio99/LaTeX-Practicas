		\begin{tikzpicture}
			\begin{groupplot}[group style={group size=2 by 1, horizontal sep=2cm},

				\nextgroupplot[title={$x^2$}]
				\addplot[blue, thick, domain=-2:2]{x^2};
				\nextgroupplot[title={$e^{-x}$}]

			\end{groupplot}
		\end{tikzpicture}
	\end{center}
	
	% ===================== fillbetween =====================
	\section*{Librería: fillbetween}
	\begin{center}
		\begin{tikzpicture}
			\begin{axis}[grid=both, xmin=0, xmax=2]
				\addplot[name path=A, domain=0:2]{x^2};
				\addplot[name path=B, domain=0:2]{2*x};
				\addplot[blue!20] fill between[of=A and B];
			\end{axis}
		\end{tikzpicture}
	\end{center}
